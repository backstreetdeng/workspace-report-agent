prose
# 汽车市场战略决策智能体 - OpenProse 标准语法版
# 使用 use + do 语法调用 skill
# 版本：v4.0

## start

# 变量定义
let user_question = {{user.input}}
let result = ""

print "🚗 汽车市场战略智能体启动..."
print "问题：{{user_question}}"

# ======================================
# 阶段1：意图识别
# ======================================
print "【意图识别中...】"

const intent_result = do intent-classifier.classify
  question: {{user_question}}
end

let intent_type = intent_result.intent_type
let confidence = intent_result.confidence
let brands_mentioned = intent_result.brands_mentioned

print "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
print "意图类型：{{intent_type}} (置信度：{{confidence}})"
print "品牌：{{brands_mentioned}}"
print "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

# ======================================
# 阶段2：数据检索
# ======================================
print "【数据检索中...】"

# 向量检索
const vector_result = do pg-vector-search.search
  query: {{user_question}}
  top_k: 6
  search_mode: hybrid
end

let vector_count = vector_result.count

# 结构化查询
const sql_result = do nl2sql-pg.query
  question: {{user_question}}
  execute: true
end

let sql_count = sql_result.record_count

print "向量知识：{{vector_count}} 条"
print "结构化数据：{{sql_count}} 条"

# ======================================
# 阶段3：四大框架分析
# ======================================
print "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
print "【执行四大框架分析】"
print "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

# PEST分析
const pest_result = do automotive-strategy-analysis.pest
end

# 波特五力
const porter_result = do automotive-strategy-analysis.porter
end

# SWOT分析（如果有品牌）
if {{brands_mentioned}} is not empty:
    const swot_result = do automotive-strategy-analysis.swot
      brand: {{brands_mentioned.[0]}}
    end

    # 4P分析
    const fourp_result = do automotive-strategy-analysis.fourp
      brand: {{brands_mentioned.[0]}}
    end

    let swot_data = swot_result.data
    let fourp_data = fourp_result.data
else:
    let swot_data = null
    let fourp_data = null
    print "【未识别品牌，跳过SWOT/4P】"
end

# ======================================
# 阶段4：报告生成
# ======================================
print "【生成报告中...】"

const report_result = do report-generator.generate
  question: {{user_question}}
  intent_type: {{intent_type}}
  pest_result: {{pest_result.data}}
  porter_result: {{porter_result.data}}
  swot_result: {{swot_data}}
  fourp_result: {{fourp_data}}
end

let final_report = report_result.markdown

# ======================================
# 阶段5：输出结果
# ======================================
print "╔══════════════════════════════════════════════════════════════╗"
print "║          汽车市场战略分析报告                          ║"
print "╠══════════════════════════════════════════════════════════════╣"
print "{{final_report}}"
print "╚══════════════════════════════════════════════════════════════╝"

# ======================================
# 阶段6：记忆归档
# ======================================
save_memory
content:
  时间：{{system.time}}
  用户问题：{{user_question}}
  意图类型：{{intent_type}}
  置信度：{{confidence}}
end
